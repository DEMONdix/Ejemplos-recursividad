"""
Método que conviete un número a binarios 
"""
def binario(n):
    if n/2 == 0:
        return n
    else:
        return (n%2) + 10 * binario(n//2)

"""
Metodo que retorna el modulo de dos valores
"""
def mod(a,b):
    if a<b:
        return a
    else:
        return mod(a-b,b)
